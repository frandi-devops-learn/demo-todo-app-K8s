# Install ALB via helm
1. helm repo add eks https://aws.github.io/eks-charts

1.1. helm repo update

1.2. helm install aws-load-balancer-controller eks/aws-load-balancer-controller -n kube-system --set clusterName=microk8s-cluster --set serviceAccount.create=false --set region=ap-southeast-1 --set vpcId=vpc-07bedbd5b01030303
---------------

# Install ArgoCD
2. kubectl create namespace argocd

2.1. helm repo add argo https://argoproj.github.io/argo-helm

2.2. helm install argocd argo/argo-cd --namespace argocd --set server.service.type=NodePort --set server.extraArgs={--insecure}

2.3. kubectl -n argocd get secret argocd-initial-admin-secret -o jsonpath="{.data.password}" | base64 -d

---------------

# Install sealed secret
3. helm repo add sealed-secrets https://bitnami-labs.github.io/sealed-secrets

3.1. helm repo update

3.2. helm install sealed-secrets sealed-secrets/sealed-secrets --namespace kube-system --set fullnameOverride=sealed-secrets-controller

3.3. kubeseal --fetch-cert > sealed-secrets-public-key.pem

3.4. kubectl create secret generic demo-todo-secret --namespace demo-todo --from-literal DATABASE_URL=postgresql://username:password@yourdburl:5432/uatdb --from-literal TELEGRAM_BOT_TOKEN=yourtgtoken --from-literal TELEGRAM_CHAT_ID=yourtgchatid --dry-run=client -oyaml > sec.yaml

3.5. kubeseal --format=yaml --cert=sealed-secrets-public-key.pem < sec.yaml > sealed-secret.yaml

---------------

# Install ArgoCD Image Updater

4. helm install argocd-image-updater argo/argocd-image-updater -n argocd

---------------

# Install vault
4. helm repo add hashicorp https://helm.releases.hashicorp.com

4.1. helm repo update

4.2. helm install vault hashicorp/vault -n vault --create-namespace -f values.yaml

4.3. kubectl -n vault exec -it vault-0 -- vault operator init

4.4. kubectl -n vault exec -it vault-0 -- vault operator unseal ( 3 times )

4.5. kubectl -n vault exec -it vault-0 -- vault status

4.6. kubectl exec -n vault vault-0 -- vault operator init -key-shares=5 -key-threshold=3 -format=json > ~/vault/cluster-keys.json
